### 递归实现   
```javascript   
let reverseList = (head) => {
        if (!head) {
          return null;
        }
        let reverse = (pre, cur) => {
          if (cur) {
            // 保存 next 节点
            let next = cur.next;
            cur.next = pre;
            return reverse(cur, next);
          }
          return pre;
        };
        return reverse(null, head);
      };   
      ```   
### 循环   
```javascript   
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
let reverseList = (head) => {
        if (!head) {
          return null;
        }
        let pre = null;
        cur = head;
        while (cur) {
          //保存next
          let next = cur.next;
          //替换next
          cur.next = pre;
          //设置pre的值
          pre = cur;
          //设置当前项的值
          cur = next;
        }
      return pre;
      };
```
